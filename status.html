# GUI for Status Update This is a simple GUI application created with Python's tkinter module. The purpose of this application is to allow a user to display their current status (Busy, Free, or Busy Send Text) by clicking a button. After a button is clicked, the status is displayed on the screen and the status buttons disappear, replaced by a Reset button. The Reset button will clear the displayed status and bring back the status buttons. ## Requirements Python 3.x and tkinter module. ## Code ```python import tkinter as tk def update_status(color, text, fontcolor): root.configure(background=color) status_label.config(text=text, fg=fontcolor, bg=color, font=("Helvetica", 12, "bold")) red_button.pack_forget() green_button.pack_forget() yellow_button.pack_forget() reset_button.pack(side=tk.LEFT) def reset_status(): root.configure(background='white') status_label.config(text="", bg='white', font=("Helvetica", 12)) red_button.pack(side=tk.LEFT) green_button.pack(side=tk.LEFT) yellow_button.pack(side=tk.LEFT) reset_button.pack_forget() root = tk.Tk() root.geometry('200x200') status_label = tk.Label(root, text="", bg='white') status_label.pack() button_frame = tk.Frame(root) button_frame.pack() red_button = tk.Button(button_frame, text="Busy", command=lambda: update_status('red', 'Mommy is: Busy - emergencies only.', 'white'), height = 1, width = 10) red_button.pack(side=tk.LEFT) green_button = tk.Button(button_frame, text="Free", command=lambda: update_status('green', 'Mommy is: Available - text or talking', 'white'), height = 1, width = 10) green_button.pack(side=tk.LEFT) yellow_button = tk.Button(button_frame, text="Busy, Send Text", command=lambda: update_status('yellow', 'Mommy is: Busy- send a request or text', 'black'), height = 1, width = 10) yellow_button.pack(side=tk.LEFT) reset_button = tk.Button(button_frame, text="Reset", command=reset_status, height = 1, width = 10) root.mainloop()
